
            Procedure InsertionSort(numbers : Array[1 .. 100] of Integer; size : Integer);
            Var
            	i, j, index : Integer;


            Begin
            	For i := 2 to size-1 do
            	Begin
            		index := numbers[i];
            		j := i;
            		While ((j > 1) AND (numbers[j-1] > index)) do
            		Begin
            			numbers[j] := numbers[j-1];
            			j := j - 1;
            		End
            		numbers[j] := index;
            	End
            End
